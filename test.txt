def fill_industry_by_clientname(df):
    # Step 1: Get distinct Clientname values where Industry is missing
    distinct_clients = df[df['Industry'].isnull()].drop_duplicates(subset=['Clientname'])
    
    # Step 2: Apply IndustryCategory to create a dictionary with Clientname as keys
    industry_dict = distinct_clients.set_index('Clientname').apply(IndustryCategory, axis=1).to_dict()
    
    # Step 3: Map this dictionary back to the original DataFrame's Industry column
    df['Industry'] = df['Industry'].fillna(df['Clientname'].map(industry_dict))
    
    return df
